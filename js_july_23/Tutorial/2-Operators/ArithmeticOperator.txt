+ : addition

- : subtraction

* : multiplication

/ : division 

% : Modulus (Finds the remainder)

pre ++ : prefix Increment ( var1 = var1 + 1 )
E.g.,
=> var x = 9;
=> var y = ++x;

post ++ : postfix Increment ( var1 = var1 + 1 )
=> var x = 9;
=> var y = x++;

pre -- : prefix Decrement ( var1 = var1 - 1 )
E.g.,
=> var x = 9;
=> var y = --x;

post -- : post Decrement ( var1 = var1 - 1 )
=> var x = 9;
=> var y = x--;
